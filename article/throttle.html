# 防抖和节流

# 名词解释

防抖： 事件频繁触发时，n秒内再次被触发，则重新计算时间

# 使用场景

频繁操作点赞和取消点赞，因此需要获取最后一次操作结果并发送给服务器

# 实现原理

每次触发事件时设置一个延迟调用方法，并且取消之前的延时调用方法

# 实现代码

export function deBounce(fn, delay = 10) {
    let timer;
    return function(...args) {
        if (timer) {
            timer = clearTimeout(timer);
        }
        timer = setTimeout(() => fn.apply(this, [...args]), delay);
    };
}

# 名词解释

节流： 事件频繁触发时，n秒内只触发一次函数

# 使用场景

页面的无限加载场景下，在滚动页面时，每隔一段时间发一次 Ajax 请求，而不是在停下滚动操作时才去请求

# 实现原理

通过判断是否有延迟函数未执行

# 实现代码

export function throttle(fn, delay) {
    let flag = true;
    return function(...args) {
        if (!flag) return;
        flag = false;
        setTimeout(() => {
            fn.apply(this, [...args]);
            flag = true;
        }, delay)
    }
}
