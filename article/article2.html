递归+数组深拷贝

function copy(arr) {
    let obj = this.isArray(arr) ? [] : {};
    for (let i in arr) {
        let item = arr[i];
        if (this.isArray(item) || this.isObject(item)) {
            obj[i] = copy.call(this, arr[i])
        } else {
            obj[i] = item;
        }
    }
    return obj;
}
